Algorithm: 3. Reverse Linked List

1. How to reverse a list?

We can reverse linked list in two ways, which is iteration and recursion.

1.1 Iteration

1. We define three pointer variables: prev, cur, next and header,
prev points to the previous node, which points to NULL at the beginning of the iteration.
cur points to the current node,
next points to the next node.
header tells us where the first node is.

2. We make next point to current node's next node,
	make the current node's next pointer point to the previous node,

3. make the prev point to current node, and make the cur point the next node.

4. Repeat until the last node(the cur points to NULL).

Input: 1->2->3->4->5->NULL
Output: 5->4->3->2->1->NULL

1.2 Recursion

  1. base case: the last node, whose next is null.
  2. reverse nodes after head node recursively.

func ReverseListR(head *Node) *Node {
if nil == head.Next {
return head
}

last := ReverseListR(head.Next)
head.Next.Next = head
head.Next = nil
return last

}

last := ReverseListR(head.Next)

After ReverseListR, we get the following linked list

head.Next.Next = head

head.Next = nil

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct Node {
  int val;
  struct Node * next;
};

struct Node * reverse_list(struct Node *l) {
  if (NULL == l || NULL == l->next) {
    return l;
  }

  struct Node *cur = l;
  struct Node *prev, *next, *header;
  prev = NULL;
  while(cur) {
    header = cur;
    next = cur->next;
    cur->next = prev;
    prev = cur;
    cur = next;
  }
  return header;
}

  • Golang
package main

import "fmt"

type Node struct {
	Val  int
	Next *Node
}

func ReverseList(l *Node) *Node {
	if nil == l || nil == l.Next {
		return l
	}

	cur := l
	var prev, next, header *Node
	prev = nil
	for nil != cur {
		header = cur
		next = cur.Next
		cur.Next = prev
		prev = cur
		cur = next
	}
	return header
}

func ConstructList() *Node {
	var prev, header *Node
	for i := 1; i <= 5; i++ {
		nd := &Node{
			Val:  i,
			Next: nil,
		}
		if nil == header {
			header = nd
		}
		if nil != prev {
			prev.Next = nd
		}
		prev = nd
	}
	return header
}

func PrintList(l *Node) {
	cur := l
	for nil != cur {
		fmt.Printf("%p, val: %d, next: %p\n", cur, cur.Val, cur.Next)
		cur = cur.Next
	}
}

func main() {
	l := ConstructList()
	PrintList(l)
	l = ReverseList(l)
	PrintList(l)
}

func ReverseListR(head *Node) *Node {
	if nil == head.Next {
		return head
	}

	last := ReverseListR(head.Next)
	head.Next.Next = head
	head.Next = nil
	return last
}
